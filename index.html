const SHEET_NAME = "Tasks"; function doGet(e) { return handleRequest(e); } function doPost(e) { return handleRequest(e); } function handleRequest(e) { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName(SHEET_NAME); const action = (e.parameter && e.parameter.action) || (e.postData && JSON.parse(e.postData.contents).action); let result; if (action === "read") { result = readTasks(sheet); } else if (action === "write") { const data = JSON.parse(e.postData.contents); result = writeTasks(sheet, data.tasks); } else if (action === "addTask") { const data = JSON.parse(e.postData.contents); result = addTask(sheet, data.task); } else if (action === "updateTask") { const data = JSON.parse(e.postData.contents); result = updateTask(sheet, data.task); } else { result = { error: "Unknown action" }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } function readTasks(sheet) { const data = sheet.getDataRange().getValues(); const headers = data[0]; const tasks = data.slice(1).filter(r => r[0]).map(row => { const t = {}; headers.forEach((h, i) => t[h] = row[i] ? String(row[i]) : ""); return t; }); return { tasks }; } function writeTasks(sheet, tasks) { sheet.clearContents(); const headers = ["ID","Гарчиг","Брэнд","Ангилал","Дизайнер", "Тайлбар","Дуусах огноо","Зэрэглэл","Төлөв","Үүсгэсэн огноо"]; sheet.appendRow(headers); tasks.forEach(t => { sheet.appendRow([t.id,t.title,t.brand,t.cat,t.designer, t.desc,t.due,t.priority,t.status,t.created]); }); return { ok: true, count: tasks.length }; } function addTask(sheet, task) { sheet.appendRow([task.id,task.title,task.brand,task.cat,task.designer, task.desc,task.due,task.priority,task.status,task.created]); return { ok: true }; } function updateTask(sheet, task) { const data = sheet.getDataRange().getValues(); for (let i = 1; i < data.length; i++) { if (String(data[i][0]) === String(task.id)) { sheet.getRange(i+1,1,1,10).setValues([[ task.id,task.title,task.brand,task.cat,task.designer, task.desc,task.due,task.priority,task.status,task.created ]]); return { ok: true }; } } return { error: "Task not found" }; }
